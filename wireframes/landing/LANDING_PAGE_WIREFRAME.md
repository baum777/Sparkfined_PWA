const animateCounter = (element, target, duration) => {
  let start = 0;
  const increment = target / (duration / 16); // 60fps

  const timer = setInterval(() => {
    start += increment;
    element.textContent = Math.floor(start);
    if (start >= target) {
      element.textContent = target;

    }
  }, 16);
};
